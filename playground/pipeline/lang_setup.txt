Go
-   cd ~/
-   GO_VERSION=1.24.5
-   wget https://go.dev/dl/go${GO_VERSION}.linux-amd64.tar.gz
-   rm -rf /usr/local/go
-   tar -C /usr/local -xzf go${GO_VERSION}.linux-amd64.tar.gz
-   echo 'export PATH=$PATH:/usr/local/go/bin' >> ~/.bashrc
-   export PATH=$PATH:/usr/local/go/bin
-   go version
